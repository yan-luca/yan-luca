Sobre o Projeto

Este projeto implementa um sistema de segurança automotivo completo utilizando microcontrolador ESP32, combinando tecnologia Bluetooth e autenticação biométrica para máxima proteção veicular.

O sistema oferece múltiplas camadas de segurança, incluindo controle remoto via smartphone, acesso por impressão digital, sensores de movimento e sirene de alta potência. A solução foi projetada para ser facilmente instalada em qualquer veículo.

Principais características:

  • Controle remoto via Bluetooth (alcance de 100m)
  • Autenticação por impressão digital
  • Sensores de abertura de portas e capô
  • Sensor de movimento interno
  • Sirene de 120dB com bateria backup
  • LED indicador de status
  • Modo valet para manobristas
  • Configuração via aplicativo móvel

#include <BluetoothSerial.h> #include <Adafruit_Fingerprint.h> #include <EEPROM.h> BluetoothSerial SerialBT; Adafruit_Fingerprint finger = Adafruit_Fingerprint(&Serial2); // Pinos do sistema const int SIREN_PIN = 2; const int LED_STATUS = 5; const int DOOR_SENSOR = 18; const int HOOD_SENSOR = 19; const int MOTION_SENSOR = 21; const int RELAY_IGNITION = 22; // Estados do sistema enum SystemState { DISARMED, ARMED, TRIGGERED, VALET_MODE }; SystemState currentState = DISARMED; bool doorOpen = false; bool motionDetected = false; unsigned long lastActivity = 0; void setup() { Serial.begin(115200); SerialBT.begin("CarAlarm_ESP32"); // Inicializar sensor biométrico finger.begin(57600); // Configurar pinos pinMode(SIREN_PIN, OUTPUT); pinMode(LED_STATUS, OUTPUT); pinMode(DOOR_SENSOR, INPUT_PULLUP); pinMode(HOOD_SENSOR, INPUT_PULLUP); pinMode(MOTION_SENSOR, INPUT); pinMode(RELAY_IGNITION, OUTPUT); // Carregar configurações da EEPROM loadSettings(); Serial.println("Sistema de Alarme Iniciado"); blinkStatus(3); // 3 piscadas = sistema online } void loop() { checkBluetoothCommands(); checkSensors(); checkBiometric(); updateStatusLED(); // Anti-tampering - verifica se o sistema foi desconectado if (millis() - lastActivity > 300000) { // 5 minutos triggerTamperAlarm(); } delay(100); } void checkSensors() { bool newDoorState = !digitalRead(DOOR_SENSOR); bool newHoodState = !digitalRead(HOOD_SENSOR); bool newMotionState = digitalRead(MOTION_SENSOR); if (currentState == ARMED) { if (newDoorState || newHoodState || newMotionState) { triggerAlarm("Sensor ativado"); } } doorOpen = newDoorState; motionDetected = newMotionState; } void triggerAlarm(String reason) { currentState = TRIGGERED; // Ativar sirene digitalWrite(SIREN_PIN, HIGH); // Enviar notificação via Bluetooth SerialBT.println("ALARM:" + reason); // Log do evento logEvent("ALARM_TRIGGERED", reason); // Sirene por 30 segundos delay(30000); digitalWrite(SIREN_PIN, LOW); currentState = ARMED; // Voltar ao estado armado } int getBiometricID() { uint8_t p = finger.getImage(); if (p != FINGERPRINT_OK) return -1; p = finger.image2Tz(); if (p != FINGERPRINT_OK) return -1; p = finger.fingerFastSearch(); if (p != FINGERPRINT_OK) return -1; return finger.fingerID; }
